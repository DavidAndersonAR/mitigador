## API Client Surface

`web/src/api/traffic.ts` — JSON shapes expected from Plan 03 handlers:

```typescript
// GET /api/traffic/top20
interface TopTalkersResponse {
  items: TopTalker[];     // up to 20, sorted bps desc
  generated_at: string;  // RFC3339
}
interface TopTalker {
  ip: string;
  hostgroup: string | null;
  bps: number;
  pps: number;
  dominant_proto: 'udp' | 'icmp' | 'other';
}

// GET /api/traffic/host/{ip}  (404 if unknown)
interface HostTrafficResponse {
  ip: string;
  hostgroup: string | null;
  generated_at: string;
  buckets: HostBucket[];  // newest-first, index 0 = now
}
interface HostBucket {
  offset_s: number;
  pps: number; bps: number;
  pps_udp: number; bps_udp: number;
  pps_icmp: number; bps_icmp: number;
  pps_other: number; bps_other: number;
}
```
