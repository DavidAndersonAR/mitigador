```ts
// .setData(data, resetScales?): void
// .destroy(): void
// .setSize({ width, height }): void

// AlignedData layout for stacked area with 3 series:
//   [
//     xValues,            // x axis: [-59, -58, ..., -1, 0]  (seconds ago, newest right)
//     udpCumulative,      // y series 1 (bottom of stack)
//     udpPlusIcmp,        // y series 2 (UDP + ICMP)
//     udpPlusIcmpPlusOther // y series 3 (full total — drawn on top)
//   ]
//
// uPlot draws each series independently; to get a "stacked area" look we feed cumulative
// sums and use band: {} between consecutive series. See uPlot demo "stacked-area" for
// the canonical pattern: https://leeoniya.github.io/uPlot/demos/stacked-bars.html
```

    ```vue
    <script setup lang="ts">
    import { ref, watch, onMounted, onUnmounted, onBeforeUnmount } from 'vue';
    import { useI18n } from 'vue-i18n';
    import { NEmpty } from 'naive-ui';
    import uPlot from 'uplot';
    import 'uplot/dist/uPlot.min.css';
    import type { HostBucket } from '@/api/traffic';

    // Series colors picked from Naive UI dark theme palette so the chart fits the existing
    // dashboard. Documented hex values (do NOT change without updating DashboardView too):
    //   UDP   → #d03050 (red — same as the active-attack accent in DashboardView)
    //   ICMP  → #f0a020 (amber — same as the medium-confidence accent)
    //   Other → #18a058 (green — same as the success accent)
    // Fill is the stroke at 28% opacity for the stacked-area look.
    const COLOR_UDP_STROKE = '#d03050';
    const COLOR_UDP_FILL = 'rgba(208, 48, 80, 0.28)';
    const COLOR_ICMP_STROKE = '#f0a020';
    const COLOR_ICMP_FILL = 'rgba(240, 160, 32, 0.28)';
    const COLOR_OTHER_STROKE = '#18a058';
    const COLOR_OTHER_FILL = 'rgba(24, 160, 88, 0.28)';
    const COLOR_AXIS = '#a0a0a0';
    const COLOR_GRID = '#2a2a30';

    const props = defineProps<{
      buckets: HostBucket[];     // expected length 60; index 0 = "now"
      mode: 'bps' | 'pps';        // D-08 toggle
      height?: number;            // optional; default 280
    }>();

    const { t } = useI18n();

    const containerRef = ref<HTMLDivElement | null>(null);
    let chart: uPlot | null = null;
    let ro: ResizeObserver | null = null;

    function formatYAxisValue(v: number | null | undefined): string {
      if (v == null) return '';
      if (props.mode === 'bps') {
        if (v >= 1e9) return (v / 1e9).toFixed(1) + ' Gbps';
        if (v >= 1e6) return (v / 1e6).toFixed(1) + ' Mbps';
        if (v >= 1e3) return (v / 1e3).toFixed(1) + ' Kbps';
        return v.toFixed(0) + ' bps';
      }
      if (v >= 1e6) return (v / 1e6).toFixed(1) + 'M';
      if (v >= 1e3) return (v / 1e3).toFixed(1) + 'K';
      return v.toFixed(0);
    }

    // Build uPlot AlignedData from props.buckets.
    //
    // Bucket index 0 = newest. uPlot expects X ascending left→right, so we flip the array
    // to put "now" on the right edge (more intuitive: time flows L→R).
    //
    // Stacked-area encoding: y1 = udp, y2 = udp+icmp, y3 = udp+icmp+other.
    function buildData(): uPlot.AlignedData {
      const n = props.buckets.length;
      const xs: number[] = new Array(n);
      const udp: number[] = new Array(n);
      const udpIcmp: number[] = new Array(n);
      const total: number[] = new Array(n);
      // i in source = newest-first; flipped index = oldest-first.
      for (let i = 0; i < n; i++) {
        const src = props.buckets[n - 1 - i];
        // X axis: seconds offset from "now"; rightmost point = 0.
        xs[i] = -src.offset_s;
        const u = props.mode === 'bps' ? src.bps_udp : src.pps_udp;
        const ic = props.mode === 'bps' ? src.bps_icmp : src.pps_icmp;
        const ot = props.mode === 'bps' ? src.bps_other : src.pps_other;
        udp[i] = u;
        udpIcmp[i] = u + ic;
        total[i] = u + ic + ot;
      }
      return [xs, udp, udpIcmp, total];
    }

    function buildOptions(width: number, height: number): uPlot.Options {
      const tSeriesUDP = t('traffic.chart.series.udp');
      const tSeriesICMP = t('traffic.chart.series.icmp');
      const tSeriesOther = t('traffic.chart.series.other');
      const tAxisTime = t('traffic.chart.axis.time');
      const tAxisY = props.mode === 'bps' ? t('traffic.chart.axis.bps') : t('traffic.chart.axis.pps');
      return {
        width,
        height,
        cursor: { drag: { x: false, y: false }, points: { size: 6 } },
        legend: { show: false },
        scales: { x: { time: false }, y: { auto: true } },
        axes: [
          {
            stroke: COLOR_AXIS,
            grid: { stroke: COLOR_GRID, width: 1 },
            ticks: { stroke: COLOR_GRID },
            label: tAxisTime,
            labelGap: 6,
            values: (_u, splits) => splits.map((s) => (s === 0 ? '0' : String(s) + 's')),
          },
          {
            stroke: COLOR_AXIS,
            grid: { stroke: COLOR_GRID, width: 1 },
            ticks: { stroke: COLOR_GRID },
            label: tAxisY,
            labelGap: 6,
            values: (_u, splits) => splits.map(formatYAxisValue),
          },
        ],
        series: [
          {}, // x — required placeholder
          {
            label: tSeriesUDP,
            stroke: COLOR_UDP_STROKE,
            fill: COLOR_UDP_FILL,
            width: 1.5,
            value: (_u, v) => formatYAxisValue(v ?? 0),
          },
          {
            label: tSeriesICMP,
            stroke: COLOR_ICMP_STROKE,
            fill: COLOR_ICMP_FILL,
            width: 1.5,
            value: (_u, v) => formatYAxisValue(v ?? 0),
          },
          {
            label: tSeriesOther,
            stroke: COLOR_OTHER_STROKE,
            fill: COLOR_OTHER_FILL,
            width: 1.5,
            value: (_u, v) => formatYAxisValue(v ?? 0),
          },
        ],
        bands: [
          // Fill between adjacent series for the stacked-area effect.
          { series: [2, 1] }, // band between icmp-stack and udp-stack
          { series: [3, 2] }, // band between total-stack and icmp-stack
        ],
      };
    }

    function mountChart() {
      if (!containerRef.value) return;
      const w = containerRef.value.clientWidth || 600;
      const h = props.height ?? 280;
      const data = buildData();
      const opts = buildOptions(w, h);
      chart = new uPlot(opts, data, containerRef.value);
    }

    function destroyChart() {
      if (chart) { chart.destroy(); chart = null; }
    }

    function handleResize() {
      if (chart && containerRef.value) {
        chart.setSize({ width: containerRef.value.clientWidth, height: props.height ?? 280 });
      }
    }

    onMounted(() => {
      if (props.buckets.length === 0) return; // empty state handled by template
      mountChart();
      if (typeof ResizeObserver !== 'undefined' && containerRef.value) {
        ro = new ResizeObserver(handleResize);
        ro.observe(containerRef.value);
      }
    });

    // Re-init only when `mode` flips (axis label + scale unit change requires new opts).
    // For buckets-only changes we use setData to avoid the flicker of a full re-init.
    watch(() => props.mode, () => {
      destroyChart();
      if (props.buckets.length > 0) mountChart();
    });

    watch(() => props.buckets, () => {
      if (props.buckets.length === 0) {
        destroyChart();
        return;
      }
      if (!chart) { mountChart(); return; }
      chart.setData(buildData());
    }, { deep: true });

    onBeforeUnmount(() => {
      if (ro && containerRef.value) ro.unobserve(containerRef.value);
      ro = null;
    });

    onUnmounted(() => {
      destroyChart();
    });
    </script>

    <template>
      <div class="host-traffic-chart">
        <div v-if="buckets.length === 0" class="empty-state">
          <NEmpty :description="t('traffic.chart.empty')" />
        </div>
        <div v-else ref="containerRef" class="chart-container" :style="{ height: (height ?? 280) + 'px' }"></div>
      </div>
    </template>

    <style scoped>
    .host-traffic-chart {
      width: 100%;
    }
    .chart-container {
      width: 100%;
    }
    .empty-state {
      min-height: 200px;
      display: flex;
      align-items: center;
      justify-content: center;
    }
    /* uPlot's stylesheet is dark-theme-agnostic; nudge text color for legibility. */
    :deep(.u-legend) { color: #c0c0c0; }
    :deep(.u-title) { color: #e0e0e0; }
    </style>
    ```

    Implementation notes for the executor:
    - DO use the project's existing dark-theme hex values (`#1c1c21`, `#a0a0a0`, `#e0e0e0`, `#2a2a30`). The three series colors are deliberately the SAME three accents the DashboardView already uses for high/medium/low confidence — keeps the visual language consistent.
    - DO mount uPlot with the parent's `clientWidth` and use a `ResizeObserver` to handle drawer width changes (the drawer in Plan 05 may resize on viewport change).
    - DO destroy on unmount (`chart.destroy()`) AND disconnect the ResizeObserver. uPlot leaks event listeners on `window` if you don't call destroy.
    - DO NOT use `uplot/dist/uPlot.iife.min.js` — use the ESM build (`import uPlot from 'uplot'`); Vite handles the rest.
    - DO NOT add a tooltip plugin in this plan. uPlot's built-in cursor + `series[i].value` formatter is sufficient for v1 (D-discretion). Future tooltip plugins can wrap this component.
    - DO NOT call `fetchHostTraffic` from this component. It is data-agnostic; Plan 05 owns fetching.
    - The `bands` config for stacked area: `{ series: [aboveIdx, belowIdx] }` fills between two series. We use cumulative sums so each band is the "delta" between layers.
    - If the executor runs into uPlot's `AlignedData` type complaining about `number[]` vs `(number | null)[]`, cast at the return: `return [xs, udp, udpIcmp, total] as uPlot.AlignedData`.
    - The `min-h:280` default keeps the chart legible inside a drawer. The Plan-05 view can override via `:height="320"` etc.
