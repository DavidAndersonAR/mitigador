    ```go
    package api

    import (
        "net/http"
        "net/netip"
        "time"

        "github.com/go-chi/chi/v5"

        "github.com/mitigador/mitigador/internal/aggregate"
    )

    // maxIPLen guards against absurd path-param input before parsing.
    // Longest textual IPv6 (with embedded IPv4) is 45 chars; we cap at 64 for safety.
    const maxIPLen = 64

    type topTalkerItem struct {
        IP            string  `json:"ip"`
        Hostgroup     *string `json:"hostgroup"`
        Bps           uint64  `json:"bps"`
        Pps           uint64  `json:"pps"`
        DominantProto string  `json:"dominant_proto"`
    }

    type top20Payload struct {
        Items       []topTalkerItem `json:"items"`
        GeneratedAt time.Time       `json:"generated_at"`
    }

    type hostBucketItem struct {
        OffsetS  int    `json:"offset_s"`
        Pps      uint64 `json:"pps"`
        Bps      uint64 `json:"bps"`
        PpsUDP   uint64 `json:"pps_udp"`
        BpsUDP   uint64 `json:"bps_udp"`
        PpsICMP  uint64 `json:"pps_icmp"`
        BpsICMP  uint64 `json:"bps_icmp"`
        PpsOther uint64 `json:"pps_other"`
        BpsOther uint64 `json:"bps_other"`
    }

    type hostPayload struct {
        IP          string           `json:"ip"`
        Hostgroup   *string          `json:"hostgroup"`
        GeneratedAt time.Time        `json:"generated_at"`
        Buckets     []hostBucketItem `json:"buckets"`
    }

    // resolveHostgroup returns the longest-prefix-match hostgroup name (or nil).
    // Defensive: returns nil if Catalog is nil (avoids panic in tests that don't wire it).
    func resolveHostgroup(deps Deps, ip netip.Addr) *string {
        if deps.Catalog == nil {
            return nil
        }
        matches := deps.Catalog.Lookup(ip)
        if len(matches) == 0 {
            return nil
        }
        name := matches[0].HostgroupName
        return &name
    }

    // handleTrafficTop20 handles GET /api/traffic/top20.
    // Returns the top 20 hosts by bps over the current 60s window (D-03, D-04, D-05).
    // Always 200, never 404 — empty store yields {"items":[], "generated_at":…}.
    func handleTrafficTop20(deps Deps) http.HandlerFunc {
        return func(w http.ResponseWriter, r *http.Request) {
            if deps.Store == nil {
                writeJSON(w, http.StatusInternalServerError, map[string]string{"error": "internal"})
                return
            }
            now := time.Now()
            entries := deps.Store.Top(now.Unix(), 20)
            items := make([]topTalkerItem, len(entries))
            for i, e := range entries {
                items[i] = topTalkerItem{
                    IP:            e.IP.String(),
                    Hostgroup:     resolveHostgroup(deps, e.IP),
                    Bps:           e.Bps,
                    Pps:           e.Pps,
                    DominantProto: e.DominantProto,
                }
            }
            writeJSON(w, http.StatusOK, top20Payload{Items: items, GeneratedAt: now.UTC()})
        }
    }

    // handleTrafficHost handles GET /api/traffic/host/{ip}.
    // Validation:
    //   - {ip} must be ≤ maxIPLen chars (T-01.1-02: log-injection / DoS mitigation)
    //   - {ip} must parse via netip.ParseAddr; otherwise 400 invalid_param
    // Returns 404 with no detail when the host is not in the active window — protects
    // against IP enumeration (T-01.1-03).
    func handleTrafficHost(deps Deps) http.HandlerFunc {
        return func(w http.ResponseWriter, r *http.Request) {
            if deps.Store == nil {
                writeJSON(w, http.StatusInternalServerError, map[string]string{"error": "internal"})
                return
            }
            ipStr := chi.URLParam(r, "ip")
            if len(ipStr) == 0 || len(ipStr) > maxIPLen {
                writeJSON(w, http.StatusBadRequest, map[string]string{"error": "invalid_param", "param": "ip"})
                return
            }
            addr, err := netip.ParseAddr(ipStr)
            if err != nil {
                writeJSON(w, http.StatusBadRequest, map[string]string{"error": "invalid_param", "param": "ip"})
                return
            }
            now := time.Now()
            buckets := deps.Store.Snapshot(addr, now.Unix(), aggregate.WindowSize)
            if buckets == nil {
                writeJSON(w, http.StatusNotFound, map[string]string{"error": "not_found"})
                return
            }
            items := make([]hostBucketItem, len(buckets))
            for i, b := range buckets {
                items[i] = hostBucketItem{
                    OffsetS:  i,
                    Pps:      b.Pps,
                    Bps:      b.Bps,
                    PpsUDP:   b.PpsUDP,
                    BpsUDP:   b.BpsUDP,
                    PpsICMP:  b.PpsICMP,
                    BpsICMP:  b.BpsICMP,
                    PpsOther: b.PpsOther,
                    BpsOther: b.BpsOther,
                }
            }
            writeJSON(w, http.StatusOK, hostPayload{
                IP:          addr.String(),
                Hostgroup:   resolveHostgroup(deps, addr),
                GeneratedAt: now.UTC(),
                Buckets:     items,
            })
        }
    }
    ```

    Notes:
    - Use `addr.String()` in the response so the input is normalized (e.g., `::ffff:10.0.0.1` round-trips to canonical form, and v6 addresses are lowercased). Never echo back raw `chi.URLParam` (log-injection / canonicalization mitigation — T-01.1-02).
    - Do NOT use `r.RemoteAddr` or `r.URL.Path` in any log line in this handler (information disclosure — T-01.1-06).
    - Do NOT add a slog.Info for every request (the global slogMiddleware in server.go already logs requests at a sensible level).

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → /api/traffic/* | Untrusted HTTP input crosses here. Session cookie is the only auth — no API tokens. |
| Path param `{ip}` → handler | Raw string from URL, attacker-controlled, must be parsed via `netip.ParseAddr`. |
| Handler → aggregate.Store | Trusted in-process — Store is the source of truth. No cross-tenant data because Phase 3 multi-tenant uses systemd instantiated units, not multi-org-in-one-binary. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01.1-01 | Spoofing / Auth bypass | Both routes | mitigate | Routes mounted INSIDE `r.Group(requireAuth + csrfMiddleware)` in `server.go`. Verified by `TestTraffic_Top20_RequiresAuth` and `TestTraffic_Host_RequiresAuth` — anonymous GET → 401. |
| T-01.1-02 | Tampering / Injection (log + canonicalization) | `{ip}` path param | mitigate | Length cap (≤ 64 chars) BEFORE parse; reject anything `netip.ParseAddr` rejects with 400; echo back `addr.String()` (normalized form), NEVER raw chi.URLParam. Handler does not write the raw input into slog. Verified by `TestTraffic_Host_InvalidIP_Returns400` + `TestTraffic_Host_OversizeIP_Returns400`. |
| T-01.1-03 | Information disclosure / IP enumeration | `/host/{ip}` | mitigate | 404 response is identical for "never seen" vs "stale" vs "syntactically valid but never in store". No timing oracle attempted — the constant-time guarantee is out of scope for an internal ops tool but the equal-response is enforced. |
| T-01.1-04 | DoS / amplification via polling cadence | `/top20` | accept | 1s polling per browser is the FE design (D-17). Handler is O(active_hosts × 60) memory reads under per-shard mutexes — at 1k active hosts this is ~60k uint64 reads, well below 1ms. If a future operator complains about CPU, Plan-future can move Top to a background-computed snapshot. Documented in CONTEXT.md §deferred. |
| T-01.1-05 | XSS / template injection on hostgroup name | response payload | mitigate | Hostgroup names come from operator-controlled `hostgroups.name` column in Postgres. Vue templates use `{{ }}` (escaped by default) — Plan 05 acceptance forbids `v-html` for any field from these payloads. Backend additionally returns hostgroup as a STRING (not raw HTML) — Go's `encoding/json` escapes < > & by default. |
| T-01.1-06 | Information disclosure via logs | handler logging | mitigate | Handlers do NOT call `slog.*` directly for either request input or response payloads. The global `slogMiddleware` (server.go) logs the method + URL but never headers or bodies. Top-20 IPs / per-host data therefore never hit the log stream. |
| T-01.1-07 | CSRF | non-GET methods | mitigate | Only GET methods are exposed — `csrfMiddleware` skips them by design. Mounting under `r.Group` means if a future plan adds POST/PUT/DELETE under `/api/traffic/*`, CSRF kicks in automatically. |
| T-01.1-08 | Resource exhaustion via giant path param | `{ip}` | mitigate | `maxIPLen = 64`. chi has no body parsing for GET, so request line length is the only knob; Go's `http.Server.MaxHeaderBytes` (default 1 MiB) caps the URL.
