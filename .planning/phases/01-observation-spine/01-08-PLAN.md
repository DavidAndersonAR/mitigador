package incident

import (
    "context"
    "net/netip"
    "time"

    "github.com/jackc/pgx/v5/pgxpool"

    "github.com/mitigador/mitigador/internal/detect"
)

type Incident struct {
    ID          string
    HostIP      netip.Addr
    Vector      detect.Vector
    Hostgroup   string
    StartedAt   time.Time
    EndedAt     *time.Time // nil = still active
    PeakPps     uint64
    PeakBps     uint64
    Score       float64
}

type Update struct {
    IncidentID  string
    ObservedAt  time.Time
    Pps         uint64
    Bps         uint64
    Score       float64
    Kind        string // "started" | "update" | "ended"
}

// Filter narrows List results.
type Filter struct {
    HostIP    *netip.Addr
    Vector    *detect.Vector
    Since     *time.Time
    Until     *time.Time
    ActiveOnly bool
    Limit     int  // default 50, max 500
    Offset    int
}

// ListResult includes pagination metadata.
type ListResult struct {
    Items     []Incident
    Total     int64
}

type Store struct { /* pool */ }

func NewStore(pool *pgxpool.Pool) *Store

// Create inserts an incidents row + a started attack_updates row, atomically.
func (s *Store) Create(ctx context.Context, ev detect.AttackEvent) error

// Update inserts an attack_updates row and bumps peak_pps/bps in incidents if needed.
func (s *Store) Update(ctx context.Context, ev detect.AttackEvent) error

// End updates incidents.ended_at + peaks + score and inserts a final attack_updates row, atomically.
func (s *Store) End(ctx context.Context, ev detect.AttackEvent) error

func (s *Store) List(ctx context.Context, f Filter) (*ListResult, error)
func (s *Store) Get(ctx context.Context, id string) (*Incident, []Update, error)

       package incident

       import (
           "context"
           "errors"
           "fmt"
           "net/netip"
           "time"

           "github.com/jackc/pgx/v5"
           "github.com/jackc/pgx/v5/pgxpool"

           "github.com/mitigador/mitigador/internal/detect"
       )

       type Incident struct {
           ID        string
           HostIP    netip.Addr
           Vector    detect.Vector
           Hostgroup string
           StartedAt time.Time
           EndedAt   *time.Time
           PeakPps   uint64
           PeakBps   uint64
           Score     float64
       }

           IncidentID string
           ObservedAt time.Time
           Pps        uint64
           Bps        uint64
           Score      float64
           Kind       string
       }

       type Filter struct {
           HostIP     *netip.Addr
           Vector     *detect.Vector
           Since      *time.Time
           Until      *time.Time
           ActiveOnly bool
           Limit      int
           Offset     int
       }

       type ListResult struct {
           Items []Incident
           Total int64
       }

       var ErrNotFound = errors.New("incident: not found")

       type Store struct { pool *pgxpool.Pool }

       func NewStore(pool *pgxpool.Pool) *Store { return &Store{pool: pool} }

       func (s *Store) Create(ctx context.Context, ev detect.AttackEvent) error {
           tx, err := s.pool.Begin(ctx)
           if err != nil { return fmt.Errorf("incident: begin: %w", err) }
           defer tx.Rollback(ctx)
           _, err = tx.Exec(ctx, `
               INSERT INTO incidents (id, host_ip, vector, hostgroup_id, started_at, peak_pps, peak_bps, score, details)
               VALUES ($1, $2::inet, $3, (SELECT id FROM hostgroups WHERE name=$4), $5, $6, $7, $8, '{}'::jsonb)
               ON CONFLICT (id) DO NOTHING
           `, ev.IncidentID, ev.HostIP.String(), string(ev.Vector), ev.Hostgroup, ev.StartedAt, ev.PeakPps, ev.PeakBps, ev.Confidence)
           if err != nil { return fmt.Errorf("incident: insert: %w", err) }
           _, err = tx.Exec(ctx, `
               INSERT INTO attack_updates (incident_id, observed_at, pps, bps, score, kind)
               VALUES ($1, $2, $3, $4, $5, 'started')
           `, ev.IncidentID, ev.Now, ev.Pps, ev.Bps, ev.Confidence)
           if err != nil { return fmt.Errorf("incident: update insert: %w", err) }
           return tx.Commit(ctx)
       }

       func (s *Store) Update(ctx context.Context, ev detect.AttackEvent) error {
           tx, err := s.pool.Begin(ctx)
           if err != nil { return fmt.Errorf("incident: begin: %w", err) }
           defer tx.Rollback(ctx)
           _, err = tx.Exec(ctx, `
               UPDATE incidents
               SET peak_pps = GREATEST(peak_pps, $2),
                   peak_bps = GREATEST(peak_bps, $3),
                   score    = $4
               WHERE id=$1
           `, ev.IncidentID, ev.PeakPps, ev.PeakBps, ev.Confidence)
           if err != nil { return fmt.Errorf("incident: update incidents: %w", err) }
           _, err = tx.Exec(ctx, `
               INSERT INTO attack_updates (incident_id, observed_at, pps, bps, score, kind)
               VALUES ($1, $2, $3, $4, $5, 'update')
           `, ev.IncidentID, ev.Now, ev.Pps, ev.Bps, ev.Confidence)
           if err != nil { return fmt.Errorf("incident: insert update: %w", err) }
           return tx.Commit(ctx)
       }

       func (s *Store) End(ctx context.Context, ev detect.AttackEvent) error {
           tx, err := s.pool.Begin(ctx)
           if err != nil { return fmt.Errorf("incident: begin: %w", err) }
           defer tx.Rollback(ctx)
           _, err = tx.Exec(ctx, `
               UPDATE incidents
               SET ended_at = $2,
                   peak_pps = GREATEST(peak_pps, $3),
                   peak_bps = GREATEST(peak_bps, $4),
                   score    = $5
               WHERE id=$1
           `, ev.IncidentID, ev.EndedAt, ev.PeakPps, ev.PeakBps, ev.Confidence)
           if err != nil { return fmt.Errorf("incident: end update: %w", err) }
           _, err = tx.Exec(ctx, `
               INSERT INTO attack_updates (incident_id, observed_at, pps, bps, score, kind)
               VALUES ($1, $2, $3, $4, $5, 'ended')
           `, ev.IncidentID, ev.Now, ev.Pps, ev.Bps, ev.Confidence)
           if err != nil { return fmt.Errorf("incident: insert ended: %w", err) }
           return tx.Commit(ctx)
       }

       func (s *Store) List(ctx context.Context, f Filter) (*ListResult, error) {
           limit := f.Limit
           if limit <= 0 { limit = 50 }
           if limit > 500 { limit = 500 }

           where := "WHERE 1=1"
           args := []any{}
           idx := 1
           if f.HostIP != nil {
               where += fmt.Sprintf(" AND host_ip = $%d::inet", idx); args = append(args, f.HostIP.String()); idx++
           }
           if f.Vector != nil {
               where += fmt.Sprintf(" AND vector = $%d", idx); args = append(args, string(*f.Vector)); idx++
           }
           if f.Since != nil {
               where += fmt.Sprintf(" AND started_at >= $%d", idx); args = append(args, *f.Since); idx++
           }
           if f.Until != nil {
               where += fmt.Sprintf(" AND started_at < $%d", idx); args = append(args, *f.Until); idx++
           }
           if f.ActiveOnly {
               where += " AND ended_at IS NULL"
           }
           var total int64
           if err := s.pool.QueryRow(ctx, "SELECT COUNT(*) FROM incidents "+where, args...).Scan(&total); err != nil {
               return nil, fmt.Errorf("incident: count: %w", err)
           }
           rows, err := s.pool.Query(ctx, fmt.Sprintf(`
               SELECT i.id, host_ip::text, vector, COALESCE(h.name, ''), started_at, ended_at, peak_pps, peak_bps, score
               FROM incidents i LEFT JOIN hostgroups h ON h.id = i.hostgroup_id
               %s
               ORDER BY started_at DESC
               LIMIT $%d OFFSET $%d
           `, where, idx, idx+1), append(args, limit, f.Offset)...)
           if err != nil { return nil, fmt.Errorf("incident: list: %w", err) }
           defer rows.Close()
           var items []Incident
           for rows.Next() {
               var inc Incident
               var hostIPStr string
               var vec string
               if err := rows.Scan(&inc.ID, &hostIPStr, &vec, &inc.Hostgroup, &inc.StartedAt, &inc.EndedAt, &inc.PeakPps, &inc.PeakBps, &inc.Score); err != nil {
                   return nil, fmt.Errorf("incident: scan: %w", err)
               }
               // Strip /32 or /128 if pgx returned it.
               for i := len(hostIPStr) - 1; i >= 0; i-- {
                   if hostIPStr[i] == '/' { hostIPStr = hostIPStr[:i]; break }
               }
               ip, err := netip.ParseAddr(hostIPStr)
               if err == nil { inc.HostIP = ip }
               inc.Vector = detect.Vector(vec)
               items = append(items, inc)
           }
           return &ListResult{Items: items, Total: total}, nil
       }

       func (s *Store) Get(ctx context.Context, id string) (*Incident, []Update, error) {
           inc := &Incident{}
           var hostIPStr, vec string
           err := s.pool.QueryRow(ctx, `
               SELECT i.id, host_ip::text, vector, COALESCE(h.name, ''), started_at, ended_at, peak_pps, peak_bps, score
               FROM incidents i LEFT JOIN hostgroups h ON h.id = i.hostgroup_id
               WHERE i.id=$1
           `, id).Scan(&inc.ID, &hostIPStr, &vec, &inc.Hostgroup, &inc.StartedAt, &inc.EndedAt, &inc.PeakPps, &inc.PeakBps, &inc.Score)
           if errors.Is(err, pgx.ErrNoRows) { return nil, nil, ErrNotFound }
           if err != nil { return nil, nil, fmt.Errorf("incident: get: %w", err) }
           for i := len(hostIPStr) - 1; i >= 0; i-- {
               if hostIPStr[i] == '/' { hostIPStr = hostIPStr[:i]; break }
           }
           ip, _ := netip.ParseAddr(hostIPStr)
           inc.HostIP = ip
           inc.Vector = detect.Vector(vec)
           rows, err := s.pool.Query(ctx, `
               SELECT incident_id, observed_at, pps, bps, score, kind
               FROM attack_updates WHERE incident_id=$1 ORDER BY observed_at
           `, id)
           if err != nil { return nil, nil, fmt.Errorf("incident: updates: %w", err) }
           defer rows.Close()
           var updates []Update
           for rows.Next() {
               var u Update
               if err := rows.Scan(&u.IncidentID, &u.ObservedAt, &u.Pps, &u.Bps, &u.Score, &u.Kind); err != nil {
                   return nil, nil, fmt.Errorf("incident: scan update: %w", err)
               }
               updates = append(updates, u)
           }
           return inc, updates, nil
       }

       func (s *Store) ListActive(ctx context.Context) ([]Incident, error) {
           result, err := s.List(ctx, Filter{ActiveOnly: true, Limit: 500})
           if err != nil { return nil, err }
           return result.Items, nil
       }

       func (s *Store) CloseOrphans(ctx context.Context, cutoff time.Time) (int64, error) {
           tag, err := s.pool.Exec(ctx, `
               UPDATE incidents SET ended_at = $1
               WHERE ended_at IS NULL AND created_at < $1
           `, cutoff)
           if err != nil { return 0, fmt.Errorf("incident: close orphans: %w", err) }
           return tag.RowsAffected(), nil
       }

       package incident

       import (
           "context"
           "log/slog"
           "time"

           "github.com/mitigador/mitigador/internal/detect"
       )

       type Recorder struct {
           store *Store
           in    <-chan detect.AttackEvent
       }

       func NewRecorder(store *Store, in <-chan detect.AttackEvent) *Recorder {
           return &Recorder{store: store, in: in}
       }

       func (r *Recorder) Run(ctx context.Context) error {
           for {
               select {
               case <-ctx.Done():
                   // Drain with a short timeout.
                   drainCtx, cancel := context.WithTimeout(context.Background(), 2*time.Second)
                   r.drain(drainCtx)
                   cancel()
                   return ctx.Err()
               case ev := <-r.in:
                   r.handle(ctx, ev)
               }
           }
       }

       func (r *Recorder) drain(ctx context.Context) {
           for {
               select {
               case <-ctx.Done():
                   return
               case ev, ok := <-r.in:
                   if !ok { return }
                   r.handle(ctx, ev)
               }
           }
       }

       func (r *Recorder) handle(ctx context.Context, ev detect.AttackEvent) {
           var err error
           switch ev.State {
           case detect.StateStarted:
               err = r.store.Create(ctx, ev)
           case detect.StateUpdated:
               err = r.store.Update(ctx, ev)
           case detect.StateEnded:
               err = r.store.End(ctx, ev)
           default:
               slog.Warn("incident.recorder: unknown state", "state", string(ev.State), "incident_id", ev.IncidentID)
               return
           }
           if err != nil {
               slog.Error("incident.recorder: persist failed", "incident_id", ev.IncidentID, "state", string(ev.State), "host_ip", ev.HostIP.String(), "vector", string(ev.Vector), "err", err.Error())
           }
       }
