       ```go
       package api

       import (
           "encoding/json"
           "errors"
           "net/http"

           "github.com/alexedwards/scs/v2"

           "github.com/mitigador/mitigador/internal/user"
       )

       func handleLogin(users *user.Store, sm *scs.SessionManager) http.HandlerFunc {
           type req struct{ Username, Password string }
           return func(w http.ResponseWriter, r *http.Request) {
               var body req
               if err := json.NewDecoder(r.Body).Decode(&body); err != nil {
                   writeJSON(w, http.StatusBadRequest, map[string]string{"error":"bad_request"}); return
               }
               u, err := users.VerifyPassword(r.Context(), body.Username, body.Password)
               if err != nil {
                   if errors.Is(err, user.ErrNotFound) {
                       writeJSON(w, http.StatusUnauthorized, map[string]string{"error":"invalid_credentials"}); return
                   }
                   writeJSON(w, http.StatusUnauthorized, map[string]string{"error":"invalid_credentials"}); return
               }
               if err := sm.RenewToken(r.Context()); err != nil {
                   writeJSON(w, http.StatusInternalServerError, map[string]string{"error":"session_error"}); return
               }
               sm.Put(r.Context(), "user_id", u.ID)
               _ = users.UpdateLastLogin(r.Context(), u.ID)
               w.WriteHeader(http.StatusNoContent)
           }
       }

       func handleLogout(sm *scs.SessionManager) http.HandlerFunc {
           return func(w http.ResponseWriter, r *http.Request) {
               _ = sm.Destroy(r.Context())
               w.WriteHeader(http.StatusNoContent)
           }
       }

       func handleMe(users *user.Store, sm *scs.SessionManager) http.HandlerFunc {
           return func(w http.ResponseWriter, r *http.Request) {
               uid := sm.GetInt64(r.Context(), "user_id")
               if uid == 0 {
                   writeJSON(w, http.StatusUnauthorized, map[string]string{"error":"not_authenticated"}); return
               }
               u, err := users.GetByID(r.Context(), uid)
               if err != nil {
                   writeJSON(w, http.StatusUnauthorized, map[string]string{"error":"not_authenticated"}); return
               }
               writeJSON(w, http.StatusOK, map[string]any{
                   "id": u.ID, "username": u.Username, "email": u.Email,
                   "last_login": u.LastLogin,
               })
           }
       }

       func writeJSON(w http.ResponseWriter, status int, body any) {
           w.Header().Set("Content-Type", "application/json; charset=utf-8")
           w.WriteHeader(status)
           _ = json.NewEncoder(w).Encode(body)
       }
       ```

       ```go
       package api

       import (
           "encoding/json"
           "errors"
           "net/http"
           "net/netip"
           "strconv"
           "time"

           "github.com/go-chi/chi/v5"

           "github.com/mitigador/mitigador/internal/detect"
           "github.com/mitigador/mitigador/internal/incident"
       )

       func handleListIncidents(store *incident.Store) http.HandlerFunc {
           return func(w http.ResponseWriter, r *http.Request) {
               q := r.URL.Query()
               f := incident.Filter{}
               if v := q.Get("vector"); v != "" {
                   if v != string(detect.VectorUDPFlood) && v != string(detect.VectorICMPFlood) {
                       writeJSON(w, 400, map[string]string{"error":"invalid_param","param":"vector"}); return
                   }
                   vec := detect.Vector(v); f.Vector = &vec
               }
               if v := q.Get("host_ip"); v != "" {
                   addr, err := netip.ParseAddr(v)
                   if err != nil { writeJSON(w, 400, map[string]string{"error":"invalid_param","param":"host_ip"}); return }
                   f.HostIP = &addr
               }
               if v := q.Get("since"); v != "" {
                   t, err := time.Parse(time.RFC3339, v)
                   if err != nil { writeJSON(w, 400, map[string]string{"error":"invalid_param","param":"since"}); return }
                   f.Since = &t
               }
               if v := q.Get("until"); v != "" {
                   t, err := time.Parse(time.RFC3339, v)
                   if err != nil { writeJSON(w, 400, map[string]string{"error":"invalid_param","param":"until"}); return }
                   f.Until = &t
               }
               if q.Get("active") == "true" { f.ActiveOnly = true }
               if v := q.Get("limit"); v != "" {
                   n, err := strconv.Atoi(v)
                   if err != nil || n < 1 { writeJSON(w, 400, map[string]string{"error":"invalid_param","param":"limit"}); return }
                   f.Limit = n
               }
               if v := q.Get("offset"); v != "" {
                   n, err := strconv.Atoi(v)
                   if err != nil || n < 0 { writeJSON(w, 400, map[string]string{"error":"invalid_param","param":"offset"}); return }
                   f.Offset = n
               }
               result, err := store.List(r.Context(), f)
               if err != nil { writeJSON(w, 500, map[string]string{"error":"internal"}); return }
               // Serialize: convert netip.Addr → string for JSON.
               items := make([]map[string]any, len(result.Items))
               for i, inc := range result.Items {
                   items[i] = map[string]any{
                       "id": inc.ID, "host_ip": inc.HostIP.String(), "vector": string(inc.Vector),
                       "hostgroup": inc.Hostgroup, "started_at": inc.StartedAt,
                       "ended_at": inc.EndedAt, "peak_pps": inc.PeakPps, "peak_bps": inc.PeakBps,
                       "score": inc.Score,
                   }
               }
               writeJSON(w, 200, map[string]any{"items": items, "total": result.Total})
               _ = json.NewEncoder // silence unused if encoder isn't directly used
           }
       }

                       "peak_pps": inc.PeakPps, "peak_bps": inc.PeakBps, "score": inc.Score,

                   "updates": updates,
               })
           }
       ```

       ```go
       package api

       import (
           "context"
           "encoding/json"
           "fmt"
           "log/slog"
           "net/http"

           "github.com/mitigador/mitigador/internal/detect"
       )

       type Broker struct {
           in          <-chan detect.AttackEvent
           subscribe   chan chan ssePayload
           unsubscribe chan chan ssePayload
       }

       type ssePayload struct {
           ID    string
           Type  string
           Data  []byte
       }

       func NewBroker(in <-chan detect.AttackEvent) *Broker {
           return &Broker{
               in:          in,
               subscribe:   make(chan chan ssePayload, 4),
               unsubscribe: make(chan chan ssePayload, 4),
           }
       }

       func (b *Broker) Run(ctx context.Context) error {
           clients := map[chan ssePayload]struct{}{}
           tick := time.NewTicker(15 * time.Second)
           defer tick.Stop()
           for {
               select {
               case <-ctx.Done():
                   for c := range clients { close(c) }
                   return ctx.Err()
               case c := <-b.subscribe:
                   clients[c] = struct{}{}
               case c := <-b.unsubscribe:
                   delete(clients, c)
                   close(c)
               case ev := <-b.in:
                   evType := "attack." + string(ev.State)
                   payload, err := json.Marshal(eventToJSON(ev))
                   if err != nil { slog.Error("sse: marshal", "err", err.Error()); continue }
                   p := ssePayload{ID: ev.IncidentID, Type: evType, Data: payload}
                   for c := range clients {
                       select { case c <- p: default: }
                   }
               case <-tick.C:
                   // heartbeat: zero-value payload signals to clients to write SSE comment
                   hb := ssePayload{}
                   for c := range clients {
                       select { case c <- hb: default: }
                   }
               }
           }
       }

       func (b *Broker) Handler(w http.ResponseWriter, r *http.Request) {
           w.Header().Set("Content-Type", "text/event-stream")
           w.Header().Set("Cache-Control", "no-cache")
           w.Header().Set("Connection", "keep-alive")
           w.Header().Set("X-Accel-Buffering", "no")
           flusher, ok := w.(http.Flusher)
           if !ok { http.Error(w, "stream unsupported", 500); return }

           ch := make(chan ssePayload, 16)
           b.subscribe <- ch
           defer func() { b.unsubscribe <- ch }()

           for {
               select {
               case <-r.Context().Done(): return
               case p, ok := <-ch:
                   if !ok { return }
                   if p.ID == "" && p.Type == "" {
                       // heartbeat
                       fmt.Fprint(w, ": heartbeat\n\n")
                   } else {
                       fmt.Fprintf(w, "id: %s\nevent: %s\ndata: %s\n\n", p.ID, p.Type, p.Data)
                   }
                   flusher.Flush()
               }
           }
       }

       func eventToJSON(ev detect.AttackEvent) map[string]any {
           return map[string]any{
               "incident_id": ev.IncidentID, "state": string(ev.State),
               "host_ip":     ev.HostIP.String(), "vector": string(ev.Vector),
               "hostgroup":   ev.Hostgroup, "pps": ev.Pps, "bps": ev.Bps,
               "peak_pps":    ev.PeakPps, "peak_bps": ev.PeakBps,
               "confidence":  ev.Confidence,
               "started_at":  ev.StartedAt, "ended_at": ev.EndedAt, "now": ev.Now,
           }
       }
       ```

       ```go
       package api

       import (
           "embed"
           "io/fs"
           "net/http"
           "strings"
       )

       //go:embed all:web_dist
       var spaFS embed.FS

       // Handler returns an http.Handler that serves web_dist/* with SPA fallback to index.html.
       func staticHandler() http.Handler {
           sub, err := fs.Sub(spaFS, "web_dist")
           if err != nil {
               return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
                   http.Error(w, "SPA not built", 500)
               })
           }
           fileServer := http.FileServer(http.FS(sub))
           return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
               // Don't serve /api/* via this handler — chi already routed /api/* before us.
               if strings.HasPrefix(r.URL.Path, "/api/") {
                   http.NotFound(w, r); return
               }
               // Check if path resolves to a real file.
               name := strings.TrimPrefix(r.URL.Path, "/")
               if name == "" { name = "index.html" }
               if _, err := fs.Stat(sub, name); err != nil {
                   // SPA fallback.
                   r.URL.Path = "/"
                   indexBytes, err := fs.ReadFile(sub, "index.html")
                   if err != nil { http.NotFound(w, r); return }
                   w.Header().Set("Content-Type", "text/html; charset=utf-8")
                   _, _ = w.Write(indexBytes)
                   return
               }
               fileServer.ServeHTTP(w, r)
           })
       }
       ```
