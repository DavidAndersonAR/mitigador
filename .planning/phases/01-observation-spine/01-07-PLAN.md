       ```go
       package detect

       import "github.com/mitigador/mitigador/internal/aggregate"

       // Classify returns the dominant Vector for a window of buckets.
       // If UDP fraction > 0.5 → VectorUDPFlood.
       // Else if ICMP fraction > 0.5 → VectorICMPFlood.
       // Else returns "" (no P1 vector matches).
       func Classify(buckets []aggregate.Bucket) Vector {
           var totalPps, udpPps, icmpPps uint64
           for _, b := range buckets {
               totalPps += b.Pps
               udpPps   += b.PpsUDP
               icmpPps  += b.PpsICMP
           }
           if totalPps == 0 { return "" }
           if 2*udpPps > totalPps  { return VectorUDPFlood }
           if 2*icmpPps > totalPps { return VectorICMPFlood }
           return ""
       }
       ```

       ```go
       package detect

       import "math"

       // Confidence combines pps_ratio, bps_ratio, and duration_factor into a 0..1 score.
       // 0 at threshold; saturating toward 1 as the breach gets larger or longer.
       func Confidence(avgPps, avgBps uint64, t Threshold, durSec int) float64 {
           ppsRatio := float64(avgPps) / float64(maxU64(t.PPS, 1))
           bpsRatio := float64(avgBps) / float64(maxU64(t.BPS, 1))
           var durFactor float64
           if t.MinWindowSec > 0 {
               durFactor = math.Min(float64(durSec)/float64(t.MinWindowSec), 1.0)
           }
           score := 0.4*sat(ppsRatio) + 0.4*sat(bpsRatio) + 0.2*durFactor
           if score < 0 { return 0 }
           if score > 1 { return 1 }
           return score
       }

       func maxU64(a, b uint64) uint64 { if a > b { return a }; return b }

       func sat(x float64) float64 {
           if x <= 1 { return 0 }
           return math.Tanh(x - 1)
       }
       ```

       ```go
       package detect

       import (
           "math/rand"
           "net/netip"
           "sync"
           "time"

           "github.com/oklog/ulid/v2"
       )

       // CooldownAfterEnd is the grace period after StateEnded before the same (host, vector)
       // can re-trigger StateStarted (D-16).
       const CooldownAfterEnd = 60 * time.Second

       // UpdateAfterDuration: emit a single StateUpdated if the incident exceeds this duration (D-15).
       const UpdateAfterDuration = 5 * time.Minute

       type key struct {
           HostIP netip.Addr
           Vector Vector
       }

       // input is the per-tick observation handed to the state machine.
       type input struct {
           Now              time.Time
           CurrentPps       uint64
           CurrentBps       uint64
           AvgPpsLastWindow uint64
           AvgBpsLastWindow uint64
           Violating        bool   // pps > threshold OR bps > threshold AT THIS TICK
           DurationViolated int    // consecutive seconds violating (only meaningful when violating)
           Hostgroup        string
           Confidence       float64
       }

       type machineState int
       const (
           stateIdle machineState = iota
           stateActive
           stateCooldown
       )

       type incident struct {
           id              string
           startedAt       time.Time
           initialPeakPps  uint64
           initialPeakBps  uint64
           peakPps         uint64
           peakBps         uint64
           updateEmitted   bool
           noViolationStart time.Time // when the no-violation streak began (used for grace)
       }

       type machineEntry struct {
           state machineState
           inc   *incident
           cooldownUntil time.Time
       }

       type stateMachine struct {
           mu sync.Mutex
           m  map[key]*machineEntry
           rnd *rand.Rand
       }

       func newStateMachine() *stateMachine {
           return &stateMachine{ m: map[key]*machineEntry{}, rnd: rand.New(rand.NewSource(time.Now().UnixNano())) }
       }

       func (sm *stateMachine) newULID(now time.Time) string {
           // monotonic, lower 80 bits random
           ms := ulid.Timestamp(now)
           id, _ := ulid.New(ms, sm.rnd)
           return id.String()
       }

       // step advances the machine for one (host_ip, vector) and returns an event to emit, if any.
       // Threshold parameters are read from t for grace_sec.
       func (sm *stateMachine) step(k key, in input, t Threshold) *AttackEvent {
           sm.mu.Lock()
           defer sm.mu.Unlock()
           e, ok := sm.m[k]
           if !ok {
               e = &machineEntry{state: stateIdle}
               sm.m[k] = e
           }
           grace := time.Duration(t.GraceSec) * time.Second
           if grace == 0 { grace = 60 * time.Second }
           switch e.state {
           case stateIdle:
               if !in.Violating { return nil }
               if in.DurationViolated < t.MinWindowSec { return nil }
               // Transition to active. Emit STARTED.
               e.state = stateActive
               e.inc = &incident{
                   id:             sm.newULID(in.Now),
                   startedAt:      in.Now,
                   initialPeakPps: in.CurrentPps,
                   initialPeakBps: in.CurrentBps,
                   peakPps:        in.CurrentPps,
                   peakBps:        in.CurrentBps,
               }
               return &AttackEvent{
                   IncidentID: e.inc.id, State: StateStarted,
                   HostIP: k.HostIP, Vector: k.Vector, Hostgroup: in.Hostgroup,
                   Pps: in.CurrentPps, Bps: in.CurrentBps,
                   PeakPps: in.CurrentPps, PeakBps: in.CurrentBps,
                   Confidence: in.Confidence,
                   StartedAt: in.Now, Now: in.Now,
               }
           case stateActive:
               // Update peaks.
               if in.CurrentPps > e.inc.peakPps { e.inc.peakPps = in.CurrentPps }
               if in.CurrentBps > e.inc.peakBps { e.inc.peakBps = in.CurrentBps }

               if !in.Violating {
                   // Track no-violation streak.
                   if e.inc.noViolationStart.IsZero() { e.inc.noViolationStart = in.Now }
                   if in.Now.Sub(e.inc.noViolationStart) >= grace {
                       // ENDED.
                       endedAt := in.Now
                       ev := &AttackEvent{
                           IncidentID: e.inc.id, State: StateEnded,
                           HostIP: k.HostIP, Vector: k.Vector, Hostgroup: in.Hostgroup,
                           Pps: in.CurrentPps, Bps: in.CurrentBps,
                           PeakPps: e.inc.peakPps, PeakBps: e.inc.peakBps,
                           Confidence: in.Confidence,
                           StartedAt: e.inc.startedAt, EndedAt: endedAt, Now: in.Now,
                       }
                       e.state = stateCooldown
                       e.cooldownUntil = in.Now.Add(CooldownAfterEnd)
                       e.inc = nil
                       return ev
                   }
                   return nil
               }
               // Violating: reset no-violation streak.
               e.inc.noViolationStart = time.Time{}
               // Maybe emit one UPDATED.
               if !e.inc.updateEmitted {
                   duration := in.Now.Sub(e.inc.startedAt)
                   peakDouble := e.inc.peakPps > 2*e.inc.initialPeakPps || e.inc.peakBps > 2*e.inc.initialPeakBps
                   if duration >= UpdateAfterDuration || peakDouble {
                       e.inc.updateEmitted = true
                       return &AttackEvent{
                           IncidentID: e.inc.id, State: StateUpdated,
                           HostIP: k.HostIP, Vector: k.Vector, Hostgroup: in.Hostgroup,
                           Pps: in.CurrentPps, Bps: in.CurrentBps,
                           PeakPps: e.inc.peakPps, PeakBps: e.inc.peakBps,
                           Confidence: in.Confidence,
                           StartedAt: e.inc.startedAt, Now: in.Now,
                       }
                   }
               }
               return nil
           case stateCooldown:
               if in.Now.After(e.cooldownUntil) {
                   e.state = stateIdle
                   // Fall through to next tick handling.
               }
               return nil
           }
           return nil
       }
       ```

       ```go
       package detect

       import (
           "context"
           "log/slog"
           "sync/atomic"
           "time"

           "github.com/mitigador/mitigador/internal/aggregate"
       )

       // Engine is the 1Hz detection loop.
       type Engine struct {
           store    *aggregate.Store
           catalog  *Catalog
           out      chan<- AttackEvent
           sm       *stateMachine
           dropped  atomic.Uint64
           // per-(host, vector) consecutive violating-seconds counter.
           streak   map[key]int
       }

       // NewEngine wires the components.
       func NewEngine(store *aggregate.Store, catalog *Catalog, out chan<- AttackEvent) *Engine {
           return &Engine{store: store, catalog: catalog, out: out, sm: newStateMachine(), streak: map[key]int{}}
       }

       // Dropped returns the count of AttackEvents dropped because the out channel was full.
       func (e *Engine) Dropped() uint64 { return e.dropped.Load() }

       // Run blocks until ctx is canceled, ticking at 1 Hz.
       func (e *Engine) Run(ctx context.Context) error {
           t := time.NewTicker(time.Second)
           defer t.Stop()
           for {
               select {
               case <-ctx.Done():
                   return ctx.Err()
               case now := <-t.C:
                   e.tick(now)
               }
           }
       }

       func (e *Engine) tick(now time.Time) {
           sec := now.Unix()
           e.store.Tick(sec)
           hosts := e.store.ActiveHosts(sec)
           for _, h := range hosts {
               e.evalHost(h.IP, sec, now)
           }
       }

       func (e *Engine) evalHost(ip netip.Addr, sec int64, now time.Time) {
           thresholds := e.catalog.Lookup(ip)
           if len(thresholds) == 0 { return } // DETE-01 + D-10
           // Snapshot widest window we might need.
           maxWindow := 0
           for _, t := range thresholds {
               if t.MinWindowSec > maxWindow { maxWindow = t.MinWindowSec }
           }
           if maxWindow < 1 { maxWindow = 5 }
           buckets := e.store.Snapshot(ip, sec, maxWindow)
           if len(buckets) == 0 { return }
           vec := Classify(buckets)
           if vec == "" { return }
           t, ok := e.catalog.LookupByVector(ip, vec)
           if !ok { return }

           // Sum across t.MinWindowSec buckets.
           sumPps := uint64(0); sumBps := uint64(0)
           n := t.MinWindowSec
           if n > len(buckets) { n = len(buckets) }
           for i := 0; i < n; i++ {
               sumPps += buckets[i].Pps
               sumBps += buckets[i].Bps
           }
           avgPps := sumPps / uint64(n)
           avgBps := sumBps / uint64(n)
           violating := avgPps > t.PPS || avgBps > t.BPS

           k := key{HostIP: ip, Vector: vec}
           if violating {
               e.streak[k] = e.streak[k] + 1
           } else {
               e.streak[k] = 0
           }

           in := input{
               Now:              now,
               CurrentPps:       buckets[0].Pps,
               CurrentBps:       buckets[0].Bps,
               AvgPpsLastWindow: avgPps,
               AvgBpsLastWindow: avgBps,
               Violating:        violating,
               DurationViolated: e.streak[k],
               Hostgroup:        t.HostgroupName,
               Confidence:       Confidence(avgPps, avgBps, t, e.streak[k]),
           }
           ev := e.sm.step(k, in, t)
           if ev == nil { return }
           select {
           case e.out <- *ev:
           default:
               e.dropped.Add(1)
               slog.Warn("detect: dropped AttackEvent (bus full)", "host_ip", ip.String(), "vector", string(vec), "incident_id", ev.IncidentID)
           }
       }
       ```
